  \centering
  \begin{circuitikz}[american voltages]
    \draw
      (0,0) to[sV, l=$u_\mathrm{e}$] (0,3)
      (0,3) to[R, l=$R$] (2.5,3)
      to[L, l=$L$] (5,3)
      to[C, l=$C$, v^<=$u_C$] (5,0)
      (5,0) -- (0,0);
